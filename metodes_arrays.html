Apuntes de Métodos de Arrays en JavaScript

Métodos Comunes de Arrays

push()

Añade uno o más elementos al final del array y devuelve la nueva longitud del array. 

let frutas = ["Manzana", "Banana"];
let nuevaLongitud = frutas.push("Naranja");
console.log(frutas); // ["Manzana", "Banana", "Naranja"]
console.log(nuevaLongitud); // 3

pop()

Elimina el último elemento del array y lo devuelve. 

let ultimaFruta = frutas.pop();
console.log(frutas); // ["Manzana", "Banana"]
console.log(ultimaFruta); // "Naranja"

shift()

Elimina el primer elemento del array y lo devuelve. 

let primeraFruta = frutas.shift();
console.log(frutas); // ["Banana"]
console.log(primeraFruta); // "Manzana"

unshift()

Añade uno o más elementos al inicio del array y devuelve la nueva longitud del array. 

let nuevaLongitud = frutas.unshift("Fresa");
console.log(frutas); // ["Fresa", "Banana"]
console.log(nuevaLongitud); // 2

forEach()

Ejecuta una función para cada elemento del array. 

frutas.forEach(function(fruta) {
    console.log(fruta);
});
// "Fresa"
// "Banana"

map()

Crea un nuevo array con los resultados de la función aplicada a cada elemento. 

let frutasEnMayusculas = frutas.map(function(fruta) {
    return fruta.toUpperCase();
});
console.log(frutasEnMayusculas); // ["FRESA", "BANANA"]

filter()

Crea un nuevo array con todos los elementos que cumplan con la condición implementada por la función. 

let frutasConA = frutas.filter(function(fruta) {
    return fruta.includes("a");
});
console.log(frutasConA); // ["Banana"]

reduce()

Aplica una función a un acumulador y a cada elemento del array (de izquierda a derecha) para reducirlo a un único valor. 

let numeros = [1, 2, 3, 4];
let suma = numeros.reduce(function(acumulador, numero) {
    return acumulador + numero;
}, 0);
console.log(suma); // 10

find()

Devuelve el primer elemento del array que cumpla con la condición implementada por la función. 

let fruta = frutas.find(function(fruta) {
    return fruta.includes("a");
});
console.log(fruta); // "Banana"

findIndex()

Devuelve el índice del primer elemento del array que cumpla con la condición implementada por la función. 

let indice = frutas.findIndex(function(fruta) {
    return fruta.includes("a");
});
console.log(indice); // 1

includes()

Determina si un array incluye un determinado elemento. 

let incluyeBanana = frutas.includes("Banana");
console.log(incluyeBanana); // true

sort()

Ordena los elementos del array en su lugar y devuelve el array. 

let numeros = [4, 2, 5, 1, 3];
numeros.sort();
console.log(numeros); // [1, 2, 3, 4, 5]

reverse()

Invierte el orden de los elementos del array. 

numeros.reverse();
console.log(numeros); // [5, 4, 3, 2, 1]

concat()

Se utiliza para unir dos o más arrays. Este método no cambia los arrays existentes, sino que devuelve un nuevo array. 

let masFrutas = frutas.concat(["Naranja", "Uva"]);
console.log(masFrutas); // ["Fresa", "Banana", "Naranja", "Uva"]

slice()

Devuelve una copia de una parte del array dentro de un nuevo array. 

let algunasFrutas = frutas.slice(1, 3);
console.log(algunasFrutas); // ["Banana"]

splice()

Cambia el contenido de un array eliminando, reemplazando o añadiendo nuevos elementos. 

frutas.splice(1, 1, "Kiwi");
console.log(frutas); // ["Fresa", "Kiwi"]

Ejemplo Completo

let frutas = ["Manzana", "Banana", "Cereza"];
frutas.push("Durazno");
console.log(frutas); // ["Manzana", "Banana", "Cereza", "Durazno"]

let ultimaFruta = frutas.pop();
console.log(frutas); // ["Manzana", "Banana", "Cereza"]
console.log(ultimaFruta); // "Durazno"

frutas.forEach(function(fruta) {
    console.log(fruta);
});
// "Manzana"
// "Banana"
// "Cereza"

let frutasEnMayusculas = frutas.map(function(fruta) {
    return fruta.toUpperCase();
});
console.log(frutasEnMayusculas); // ["MANZANA", "BANANA", "CEREZA"]

let frutasConA = frutas.filter(function(fruta) {
    return fruta.includes("a");
});
console.log(frutasConA); // ["Manzana", "Banana"]

let numeros = [1, 2, 3, 4];
let suma = numeros.reduce(function(acumulador, numero) {
    return acumulador + numero;
}, 0);
console.log(suma); // 10

let fruta = frutas.find(function(fruta) {
    return fruta.includes("a");
});
console.log(fruta); // "Manzana"

let indice = frutas.findIndex(function(fruta) {
    return fruta.includes("a");
});
console.log(indice); // 0
