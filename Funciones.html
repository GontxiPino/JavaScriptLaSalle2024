Apuntes de Funciones en JavaScript

Definición de Funciones

Las funciones en JavaScript son bloques de código diseñados para realizar una tarea específica. Se ejecutan cuando algo las invoca (las llama).

Declaración de Funciones

Una función se declara con la palabra clave function, seguida de un nombre, paréntesis y un bloque de código entre llaves. 

function saludar() {
    console.log("Hola, Mundo!");
}
saludar(); // Llama a la función

Funciones con Parámetros

Las funciones pueden tomar parámetros, que son valores que se pasan a la función cuando se invoca. 

function saludar(nombre) {
    console.log("Hola, " + nombre + "!");
}
saludar("Juan"); // "Hola, Juan!"

Funciones con Valores de Retorno

Las funciones pueden devolver valores utilizando la palabra clave return. 

function sumar(a, b) {
    return a + b;
}
let resultado = sumar(5, 3);
console.log(resultado); // 8

Funciones Anónimas

Las funciones anónimas son funciones sin nombre. A menudo se asignan a una variable o se usan como argumentos para otras funciones. 

let saludar = function() {
    console.log("Hola, Mundo!");
};
saludar(); // "Hola, Mundo!"

Funciones Flecha

Las funciones flecha son una forma más concisa de escribir funciones anónimas introducidas en ES6. 

let saludar = () => {
    console.log("Hola, Mundo!");
};
saludar(); // "Hola, Mundo!"

let sumar = (a, b) => a + b;
let resultado = sumar(5, 3);
console.log(resultado); // 8

Funciones como Objetos de Primera Clase

En JavaScript, las funciones son objetos de primera clase, lo que significa que pueden ser asignadas a variables, pasadas como argumentos a otras funciones y devueltas por otras funciones. 

function operacion(a, b, func) {
    return func(a, b);
}

let resultado = operacion(5, 3, (x, y) => x * y);
console.log(resultado); // 15

Funciones Recursivas

Una función recursiva es una función que se llama a sí misma. 

function factorial(n) {
    if (n === 0) {
        return 1;
    } else {
        return n + factorial(n - 1);
    }
}
let resultado = factorial(5);
console.log(resultado); // 15

Funciones de Orden Superior

Las funciones de orden superior son funciones que pueden tomar otras funciones como argumentos y/o devolverlas como resultado. 

function aplicarOperacion(a, b, operacion) {
    return operacion(a, b);
}

let suma = (a, b) => a + b;
let multiplicacion = (a, b) => a * b;

console.log(aplicarOperacion(5, 3, suma)); // 8
console.log(aplicarOperacion(5, 3, multiplicacion)); // 15

Ejemplo Completo

// Declaración de función
function saludar(nombre) {
    console.log("Hola, " + nombre + "!");
}
saludar("Juan"); // "Hola, Juan!"

// Función con valor de retorno
function sumar(a, b) {
    return a + b;
}
let resultado = sumar(5, 3);
console.log(resultado); // 8

// Función anónima
let multiplicar = function(a, b) {
    return a * b;
};
console.log(multiplicar(4, 5)); // 20

// Función flecha
let dividir = (a, b) => a / b;
console.log(dividir(10, 2)); // 5

// Función de orden superior
function aplicarOperacion(a, b, operacion) {
    return operacion(a, b);
}
console.log(aplicarOperacion(5, 3, (x, y) => x - y)); // 2

// Función recursiva
function factorial(n) {
    if (n === 0) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}
console.log(factorial(5)); // 120
