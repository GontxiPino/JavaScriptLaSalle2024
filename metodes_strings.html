Apuntes de Métodos de Strings en JavaScript

Métodos Comunes de Strings

charAt()

Devuelve el carácter en el índice especificado. 

let texto = "JavaScript";
let caracter = texto.charAt(2);
console.log(caracter); // "v"

charCodeAt()

Devuelve el código Unicode del carácter en el índice especificado. 

let codigo = texto.charCodeAt(2);
console.log(codigo); // 118

concat()

Combina el texto de dos (o más) strings y devuelve una nueva string. 

let nuevaCadena = texto.concat(" es genial");
console.log(nuevaCadena); // "JavaScript es genial"

includes()

Determina si una string contiene los caracteres de una string especificada. 

let contiene = texto.includes("Script");
console.log(contiene); // true

indexOf()

Devuelve el índice dentro de la string de la primera aparición del valor especificado, o -1 si no se encuentra. 

let indice = texto.indexOf("Script");
console.log(indice); // 4

lastIndexOf()

Devuelve el índice dentro de la string de la última aparición del valor especificado, o -1 si no se encuentra. 

let ultimaPosicion = texto.lastIndexOf("a");
console.log(ultimaPosicion); // 3

slice()

Extrae una sección de una string y devuelve una string nueva. 

let seccion = texto.slice(0, 4);
console.log(seccion); // "Java"

split()

Divide un objeto de tipo String en un array de strings mediante la separación de la string en sub-strings. 

let palabras = texto.split("a");
console.log(palabras); // ["J", "v", "Script"]

substring()

Devuelve los caracteres de una string entre dos índices especificados. 

let subcadena = texto.substring(4, 10);
console.log(subcadena); // "Script"

toLowerCase()

Convierte una string en minúsculas. 

let minusculas = texto.toLowerCase();
console.log(minusculas); // "javascript"

toUpperCase()

Convierte una string en mayúsculas. 

let mayusculas = texto.toUpperCase();
console.log(mayusculas); // "JAVASCRIPT"

trim()

Elimina los espacios en blanco en ambos extremos de la string. 

let textoConEspacios = "   JavaScript   ";
let textoSinEspacios = textoConEspacios.trim();
console.log(textoSinEspacios); // "JavaScript"

replace()

Reemplaza una subcadena dentro de una string con otra subcadena. 

let nuevaCadena = texto.replace("Java", "Type");
console.log(nuevaCadena); // "TypeScript"

startsWith()

Determina si una string comienza con los caracteres de otra string. 

let empiezaCon = texto.startsWith("Java");
console.log(empiezaCon); // true

endsWith()

Determina si una string termina con los caracteres de otra string. 

let terminaCon = texto.endsWith("Script");
console.log(terminaCon); // true

Ejemplo Completo

let texto = "JavaScript";
console.log(texto.charAt(2)); // "v"
console.log(texto.charCodeAt(2)); // 118
console.log(texto.concat(" es genial")); // "JavaScript es genial"
console.log(texto.includes("Script")); // true
console.log(texto.indexOf("Script")); // 4
console.log(texto.lastIndexOf("a")); // 3
console.log(texto.slice(0, 4)); // "Java"
console.log(texto.split("a")); // ["J", "v", "Script"]
console.log(texto.substring(4, 10)); // "Script"
console.log(texto.toLowerCase()); // "javascript"
console.log(texto.toUpperCase()); // "JAVASCRIPT"
console.log(texto.trim()); // "JavaScript"
console.log(texto.replace("Java", "Type")); // "TypeScript"
console.log(texto.startsWith("Java")); // true
console.log(texto.endsWith("Script")); // true
